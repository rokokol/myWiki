# –£—Ä–∞–≤–Ω–µ–Ω–∏—è —Ç–µ–ø–ª–æ–ø—Ä–æ–≤–æ–¥–Ω–æ—Å—Ç–∏ —Å—Ç–µ—Ä–∂–Ω—è
–†–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ—Ç—Å—è —Å—Ç–µ—Ä–∂–µ–Ω—å –¥–ª–∏–Ω–Ω—ã $l$. –ü–æ–ª–æ–∂–∏–º $S(x)$ ‚Äî –ø–æ–ø–µ—Ä–µ—á–Ω–æ–µ —Å–µ—á–µ–Ω–∏–µ, $\rho(x)$ ‚Äî –ø–ª–æ—Ç–Ω–æ—Å—Ç—å, $C(x)$ ‚Äî —É–¥–µ–ª—å–Ω–∞—è —Ç–µ–ø–ª–æ–µ–º–∫–æ—Å—Ç—å. –û–±–æ–∑–Ω–∞—á–∏–º —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—É —á–µ—Ä–µ–∑ $u(x)$
![[—Ç–µ–ø–ª–æ–ø—Ä–æ–≤–æ–¥–Ω–æ—Å—Ç—å —Å—Ç–µ—Ä–∂–Ω—è]]

–ü—É—Å—Ç—å –≤ —Ç–æ—á–∫–µ $x_{1}$ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞ $u_{1}$, –≤ $x_{2}$ ‚Äî $u_{2}$, –ø—Ä–∏—á–µ–º $u_{2} > u_{1}$. –≠–∫—Å–ø–µ—Ä–∏–º–µ–Ω—Ç–∞–ª—å–Ω–æ –¥–æ–∫–∞–∑–∞–Ω–æ, —á—Ç–æ:
$$
u(x)=u_{1}+x \frac{u_{2}-u_{1}}{l}
$$

## –í—ã–≤–æ–¥ —É—Ä–∞–≤–Ω–µ–Ω–∏—è

> [!summary] –ó–∞–∫–æ–Ω –§—É—Ä—å–µ
> $Q(x,t)=-K(x)S(x)u_{x}(x,t)$, –≥–¥–µ $Q$ ‚Äî –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ç–µ–ø–ª–∞, $K$ ‚Äî –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç —Ç–µ–ø–ª–æ–ø—Ä–æ–≤–æ–¥–Ω–æ—Å—Ç–∏
> –≠—Ç–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ç–µ–ø–ª–∞, –ø—Ä–æ—Ç–µ–∫–∞—é—â–µ–µ —á–µ—Ä–µ–∑ —Ç–æ—á–∫—É $x$ –≤ –º–æ–º–µ–Ω—Ç –≤—Ä–µ–º–µ–Ω–∏ $t$ –≤ –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–∏ –æ—Å–∏ $Ox$

–í—ã–¥–µ–ª–∏–º –º–∞–ª—ã–π –∏–Ω—Ç–µ—Ä–≤–∞–ª $[t, t+\Delta t]$. –†–∞—Å—Å–º–æ—Ç—Ä–∏–º –ø—Ä–∏—Ä–∞—â–µ–Ω–∏–µ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—ã $u(x,t) \to u(x,t+\Delta t)$
> [!summary] –ó–∞–∫–æ–Ω —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏—è —ç–Ω–µ—Ä–≥–∏–∏ –≤ –∏–Ω—Ç–µ–≥—Ä–∞–ª—å–Ω–æ–π —Ñ–æ—Ä–º–µ
> $Q_{N}=\int\limits_{x}^{x+\Delta x}\rho(x)S(x)C(x)[u(x,t+\Delta t)-u(x,t)]dx$, –≥–¥–µ $Q_{N}$ ‚Äî –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ç–µ–ø–ª–∞ –¥–ª—è –Ω–∞–≥—Ä–µ–≤–∞–Ω–∏—è –æ—Ç $u(x,t)$ –¥–æ $u(x,t+\Delta t)$

–ü–æ—Å–º–æ—Ç—Ä–∏–º –Ω–∞ —Ä–∏—Å—É–Ω–æ–∫ –∏ –≤—ã—á–∏—Å–ª–∏–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ç–µ–ø–ª–æ—Ç—ã, –ø–æ—Å—Ç—É–ø–∞—é—â–µ–µ —Å–ª–µ–≤–∞ –∏ —Å–ø—Ä–∞–≤–∞ –ø—Ä–∏ –ø–æ–º–æ—â–∏ –∑–∞–∫–æ–Ω–∞ –§—É—Ä—å–µ:
$$
\begin{cases}
Q_{l}=-\int\limits_{t}^{t+\Delta t}K(x)S(x)u_{x}(x,t)dt \\
Q_{r}=\int\limits_{t}^{t+\Delta t}K(x+\Delta x)S(x+\Delta x)u_{x}(x+\Delta x,t)dt
\end{cases}
$$
–ê —Ç–µ–ø–µ—Ä—å –≤—ã—á–∏—Å–ª–∏–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ç–µ–ø–ª–æ—Ç—ã –≤ –∏—Å—Ç–æ—á–Ω–∏–∫–µ —Ç–µ–ø–ª–∞ (–∑–∞—à—Ç—Ä–∏—Ö–æ–≤–∞–Ω–Ω–∞—è –æ–±–ª–∞—Å—Ç—å):
$$
Q_{res}=\int\limits_{t}^{t+\Delta t}\int\limits_{x}^{x+\Delta x}S(x)f(x,t)dxdt,
$$
–≥–¥–µ $f(x,t)$ ‚Äî —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –∏—Å—Ç–æ—á–Ω–∏–∫–æ–≤ —Ç–µ–ø–ª–∞, —Ç.–µ. –∏—Ö –ø–ª–æ—Ç–Ω–æ—Å—Ç—å

–ü—É—Å—Ç—å —É –Ω–∞—Å **–Ω–µ** —Ç–µ–ø–ª–æ–∏–∑–æ–ª–∏—Ä–æ–≤–∞–Ω–Ω—ã–π –∏—Å—Ç–æ—á–Ω–∏–∫. –ü–æ—ç—Ç–æ–º—É –∏–º–µ–µ—Ç –º–µ—Å—Ç–æ —Ç–µ–ø–ª–æ–æ–±–º–µ–Ω —Å –æ–∫—Ä—É–∂–∞—é—â–µ–π —Å—Ä–µ–¥–æ–π ‚Äî –Ω–∞–π–¥–µ–º –µ–≥–æ:
$$
Q_{h.e.} = \int\limits_{t}^{t+\Delta t}\int\limits_{x}^{x+\Delta x}h(\theta(x,t)-u(x,t))dxdt,
$$
–≥–¥–µ $h$ ‚Äî –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç —Ç–µ–ø–ª–æ–æ–±–º–µ–Ω–∞, $\theta$ ‚Äî —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞ —Å—Ä–µ–¥—ã

–¢–æ–≥–¥–∞ –∏—Å–∫–æ–º–æ–µ $Q_{N}$ –±—É–¥–µ—Ç —Ä–∞–≤–Ω–æ:
$$
\begin{multline}
\int\limits_{x}^{x+\Delta x}\rho(x)S(x)C(x)[u(x,t+\Delta t)-u(x,t)]dx= \\
=\int\limits_{t}^{t+\Delta t}[K(x+\Delta x)S(x+\Delta x)u_{x}(x+\Delta x,t)-K(x)S(x)u_{x}(x,t)]dt +\\
+\int\limits_{t}^{t+\Delta t}\int\limits_{x}^{x+\Delta x}[S(x)f(x,t)+h(\theta(x,t)-u(x,t))]dxdt\\
\end{multline}
$$

–ò... –ü—Ä–∏–º–µ–Ω—è–µ–º —Ç–µ–æ—Ä–µ–º—ã –æ —Å—Ä–µ–¥–Ω–µ–º –∏ –æ –∫–æ–Ω–µ—á–Ω—ã—Ö –ø—Ä–∏—Ä–∞—â–µ–Ω–∏—è—Ö!
$$
\begin{multline}
\rho(x')S(x')C(x')u_{t}(x',t')\cancel{ \Delta x\Delta t }= \\
=[K(x'')S(x'')u_{x}(x'',t'')]_{x}\cancel{ \Delta x\Delta t }+ \\
+h(\theta(x''',t''')-u(x''',t'''))\cancel{ \Delta x\Delta t }+S(\tilde{x})f(\tilde{x},\tilde{t})\cancel{ \Delta x\Delta t }
\end{multline}
$$
–£—Å—Ç—Ä–µ–º–ª—è–µ–º $\Delta x,~\Delta t\to0$, –∫–∞–∫ –æ–±—ã—á–Ω–æ:
$$
\boxed{ \rho(x)S(x)u_{t}(x)=[K(x)S(x)u_{x}(x,t)]_{x}+h(\theta(x,t)-u(x,t))+S(x)f(x,t) },~x \in(0,l), t>0
$$
> [!tip] –ö–∞–∫ –º–æ–∂–Ω–æ –∑–∞–º–µ—Ç–∏—Ç—å, —ç—Ç–æ —É—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø–∞—Ä–∞–±–æ–ª–∏—á–µ—Å–∫–æ–≥–æ —Ç–∏–ø–∞!

### –ï—Å–ª–∏ –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã –æ–¥–Ω–æ—Ä–æ–¥–Ω—ã–µ
$$
\boxed{ u_{t}(x,t)=a^{2}u_{xx}(x,t)+\bar{f}(x,t) },~a^{2}=\frac{K}{\rho C},~\bar{f}=\frac{f}{\rho C}
$$

## –ì—Ä–∞–Ω–∏—á–Ω—ã–µ —É—Å–ª–æ–≤–∏—è
–ü–æ—Å–∫–æ–ª—å–∫—É –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–∞—è $u_{t}$ –ø–µ—Ä–≤–æ–≥–æ –ø–æ—Ä—è–¥–∫–∞, —Ç–æ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ –ª–∏—à—å –æ–¥–Ω–æ –Ω–∞—á–∞–ª—å–Ω–æ–µ —É—Å–ª–æ–≤–∏–µ ‚Äî –Ω–∞—á–∞–ª—å–Ω–∞—è —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞
### 1 —Ä–æ–¥–∞
#### –ù–µ–æ–¥–Ω–æ—Ä–æ–¥–Ω—ã–µ
–ú—ã –∑–∞–¥–∞–µ–º –∫–∞–∫—É—é-–ª–∏–±–æ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—É $\mu_{1}$ –∏ $\mu_{2}$ –Ω–∞ –∫–æ–Ω—Ü–∞—Ö —Å—Ç–µ—Ä–∂–Ω—è:
$$
\begin{cases}
u(0,t)=\mu_{1}(t) \\
u(l,t)=\mu_{2}(t)
\end{cases}
$$

#### –û–¥–Ω–æ—Ä–æ–¥–Ω—ã–µ
–ù–∞ –∫–æ–Ω—Ü–∞—Ö —Å—Ç–µ—Ä–∂–Ω—è –ø–æ–¥–¥–µ—Ä–∂–∏–≤–∞–µ—Ç—Å—è –Ω—É–ª–µ–≤–∞—è —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞:
$$
\begin{cases}
u(0,t)=0 \\
u(l,t)=0
\end{cases}
$$

### 2 —Ä–æ–¥–∞
#### –û–¥–Ω–æ—Ä–æ–¥–Ω—ã–µ
–ü—É—Å—Ç—å –∫–æ–Ω—Ü—ã —Å—Ç—Ä–µ–∂–Ω—è —Ç–µ–ø–ª–æ–∏–∑–æ–ª–∏—Ä–æ–≤–∞–Ω–Ω—ã. –¢–æ–≥–¥–∞ –Ω–∞ –Ω–∏—Ö –æ—Ç—Å—É—Ç—Å—Ç–≤—É–µ—Ç —Ç–µ–ø–ª–æ–æ–±–º–µ–Ω —Å –æ–∫—Ä—É–∂–∞—é—â–µ–π —Å—Ä–µ–¥–æ–π:
$$
\begin{multline}
\int\limits_{0}^{\Delta x}\rho(x)C(x)S(x)[u(x,t+\Delta t)-u(x,t)]dt= \\
=\int\limits_{t}^{t+\Delta t}K(\Delta x)S(\Delta x)u_{x}(\Delta x,t)dt+\int\limits_{t}^{t+\Delta t}\int\limits_{0}^{\Delta x}h(\theta(x,t)-u(x,t))dxdt
\end{multline}
$$

–¢–µ–æ—Ä–µ–º—ã –æ... —Å—Ä–µ–¥–Ω–µ–º..., –ø—Ä–∏—Ä–∞—â–µ–Ω–∏—è—Ö:
$$
\begin{multline}
\rho(x')S(x')C(x')u_{t}(x',t')\Delta x\cancel{ \Delta t }= \\
=K(\Delta x)S(\Delta x)u_{x}(\Delta x,t'')\cancel{ \Delta t }+h(\theta(x''',t''')-u(x''',t'''))\Delta x\cancel{ \Delta t }+S(x'''')f(x'''',t'''')\Delta x\cancel{ \Delta t }
\end{multline}
$$

–ü—É–º-–ø—É–º $\Delta x,~\Delta t\to0$... –ó–∞–ø–∏—à–µ–º —É—Å–ª–æ–≤–∏—è —Å—Ä–∞–∑—É –¥–ª—è 2-—Ö –∫–æ–Ω—Ü–æ–≤:
$$
\begin{cases}
\cancel{ K(0)S(0) }u_{x}(0,t)=0 \\
\cancel{ K(l)S(l) }u_{x}(l,t)=0
\end{cases}\implies
\boxed{ \begin{cases}
u_{x}(0,t)=0 \\
u_{x}(l,t)=0
\end{cases} }
$$

#### –ù–µ–æ–¥–Ω–æ—Ä–æ–¥–Ω—ã–µ
–¢–µ–ø–µ—Ä—å –∫–æ–Ω—Ü—ã –º–æ–≥—É—Ç –±—ã—Ç—å –Ω–µ —Ç–µ–ø–ª–æ–∏–∑–æ–ª–∏—Ä–æ–≤–∞–Ω–Ω—ã:
$$
\begin{multline}
\int\limits_{0}^{\Delta x}\rho(x)C(x)S(x)[u(x,t+\Delta t)-u(x,t)]dt= \\
=\int\limits_{t}^{t+\Delta t}K(\Delta x)S(\Delta x)u_{x}(\Delta x,t)dt+ \\
+\int\limits_{t}^{t+\Delta t}\int\limits_{0}^{\Delta x}h(\theta(x,t)-u(x,t))dxdt+\int\limits_{t}^{t+\Delta t}q_{l}(t)dt
\end{multline}
$$
–≥–¥–µ $q(t)$ ‚Äî —Ç–µ–ø–ª–æ–æ–±–º–µ–Ω –∫–æ–Ω—Ü–∞ —Å –æ–∫—Ä—É–∂–∞—é—â–µ–π —Å—Ä–µ–¥–æ–π

–¢–µ–æ—Ä–µ–º—ã –æ... —Å—Ä–µ–¥–Ω–µ–º..., –ø—Ä–∏—Ä–∞—â–µ–Ω–∏—è—Ö:
$$
\begin{multline}
\rho(x')S(x')C(x')u_{t}(x',t')\Delta x\cancel{ \Delta t }= \\
=K(\Delta x)S(\Delta x)u_{x}(\Delta x,t'')\cancel{ \Delta t }+h(\theta(x''',t''')-u(x''',t'''))\Delta x\cancel{ \Delta t }+ \\
+S(x'''')f(x'''',t'''')\Delta x\cancel{ \Delta t }+q_{l}(\tilde{t})\cancel{ \Delta t }
\end{multline}
$$

–ü–∞–º-–ø–∞–º $\Delta x,~\Delta t\to0$... –ó–∞–ø–∏—à–µ–º —É—Å–ª–æ–≤–∏—è —Å—Ä–∞–∑—É –¥–ª—è 2-—Ö –∫–æ–Ω—Ü–æ–≤:
$$
\boxed{ \begin{cases}
u_{x}(0,t)=-\frac{q_{l}(t)}{K(0)S(0)} \\
u_{x}(l,t)=\frac{q_{r}(t)}{K(l)S(l)}
\end{cases} }
$$

–í–æ –≤—Ç–æ—Ä–æ–º —É—Å–ª–æ–≤–∏–∏ –º–∏–Ω—É—Å —É—Ö–æ–¥–∏—Ç –∏–∑-–∑–∞ —Ç–æ–≥–æ, —á—Ç–æ $q_{r}$ –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω –ø—Ä–æ—Ç–∏–≤ –æ—Å–∏ $Ox$

### 3 —Ä–æ–¥–∞
–ù–∞ –∫–æ–Ω—Ü–∞—Ö —Å—Ç–µ—Ä–∂–Ω—è –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç –∫–æ–Ω–≤–µ–∫—Ç–∏–≤–Ω—ã–π —Ç–µ–ø–ª–æ–æ–±–º–µ–Ω –ø–æ –∑–∞–∫–æ–Ω—É –ù—å—é—Ç–æ–Ω–∞ —Å–æ —Å—Ä–µ–¥–∞–º–∏ —Å —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞–º–∏ $\theta_{1}$ –∏ $\theta_{2}$ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ. –¢–æ–≥–¥–∞ –ø–æ–ª—É—á–∞–µ–º
![[—Ç–µ–ø–ª–æ–æ–±–º–µ–Ω —Å—Ç–µ—Ä–∂–Ω—è –∫—Ä–∞–µ–≤—ã–µ —É—Å–ª–æ–≤–∏—è]]
$$
\begin{multline}
\int\limits_{0}^{\Delta x}C(x)\rho(x)S(x)[u(x, t+\Delta t)-u(x,t)]dx= \\
=\int\limits_{t}^{t+\Delta t}S(\Delta x)K(\Delta x)u_{x}(\Delta x,t)dt+\int\limits_{t}^{t+\Delta t}\int\limits_{0}^{\Delta x}S(x)f(x,t)dxdt+h\int\limits_{t}^{t+\Delta t}(\theta_{1}(t)-u(0,t))
\end{multline}
$$
–ü—Ä–∏–º–µ–Ω—è–µ–º —Ç–µ–æ—Ä–µ–º—ã –æ —Å—Ä–µ–¥–Ω–µ–º:
$$
\begin{multline}
C(x')\rho(x')S(x')u_{t}(x',t')\Delta x\cancel{ \Delta t }= \\
=S(\Delta x)K(\Delta x)u_{x}(\Delta x,t'')\cancel{ \Delta t }+S(x''')f(x''',t''')\Delta x\cancel{ \Delta t }+h(\theta_{1}(t'''')-u(0,t''''))\cancel{ \Delta t }
\end{multline}
$$
–£—Å—Ç—Ä–µ–º–∏–º $\Delta x,~\Delta t\to0$:
$$
\begin{multline}
0=S(0)K(0)u_{x}(0,t)+h(\theta_{1}(t)-u(0,t))\implies  \\
\implies -h\theta_{1}(t)=S(0)K(0)u_{x}(0,t)-hu(0,t)
\end{multline}
$$
–ê–Ω–∞–ª–æ–≥–∏—á–Ω–æ –∑–∞–ø–∏—à–µ–º –∏ –¥–ª—è –≤—Ç–æ—Ä–æ–≥–æ —É—Ä–∞–≤–Ω–µ–Ω–∏—è. –ó–∞–ø–∏—à–µ–º –∏—Ç–æ–≥–æ–≤—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç:
$$
\boxed{ \begin{cases}
-h\theta_{1}(t)=S(0)K(0)u_{x}(0,t)-hu(0,t) \\
h\theta_{2}(t)=S(l)K(l)u_{x}(l,t)+hu(l,t)
\end{cases} }
$$
